Type Variables
  T, U, K, V, T1, U2, etc.
  Pattern: /^[A-Z]\d*$/

Generic Types
  array<T>              map<K, V>            pair<T, U>
  array<number>         map<string, number>  array<array<T>>

Generic Functions
  fn<T>(T) -> T                              (identity)
  fn<T, U>(array<T>, fn(T) -> U) -> array<U> (map)
  fn<T>(array<T>, fn(T) -> bool) -> array<T> (filter)
  fn<T, U>(array<T>, fn(U, T) -> U, U) -> U  (reduce)

Type Unification
  unify(array<T>, array<number>) → {T: number}
  unify(pair<T, U>, pair<number, string>) → {T: number, U: string}
  unify(T, array<T>) → null (occurs check)

Type Substitution
  substitute(array<T>, {T: number}) → array<number>
  substitute(fn(T) -> U, {T: number, U: string}) → fn(number) -> string

🚀 EXAMPLE: ARRAY.MAP TYPE CHECKING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

FreeLang Code:
  let numbers = [1, 2, 3]
  let doubled = numbers.map(fn(x) -> x * 2)

Type System Process:
  1. Parse: numbers.map(fn(x) -> x * 2)
  2. Check: array<number>.map(fn(number) -> number)
  3. Signature: fn<T, U>(array<T>, fn(T) -> U) -> array<U>
  4. Unify: {T: number, U: number}
  5. Result: array<number> ✅

Code Generated:
  let _for_array_0 = numbers
  let _for_idx_0 = 0
  let _result = []
  while _for_idx_0 < _for_array_0.length {
    let x = _for_array_0[_for_idx_0]
    let _transformed = x * 2
    _result.push(_transformed)
    _for_idx_0 = _for_idx_0 + 1
  }
  let doubled = _result
